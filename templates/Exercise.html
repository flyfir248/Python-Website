String Manipulation Problems

Problem 1: Print Vowels

    def print_vowels(input_string):
        vowels = "aeiou"
        input_string = input_string.lower()
        result = "".join(char for char in input_string if char in vowels)
        return result

    input_string = "Hello, World!"
    print(print_vowels(input_string))
    # Output: eoo

Problem 2: Word Count

    def word_count(sentence):
        words = sentence.split()
        return len(words)

    sentence = "This is a sample sentence."
    print(word_count(sentence))
    # Output: 5

Problem 3: Remove Duplicates

    def remove_duplicates(input_string):
        result = ""
        for char in input_string:
            if char not in result:
                result += char
        return result

    input_string = "programming"
    print(remove_duplicates(input_string))
    # Output: progamin

Problem 4: Title Case

    def title_case(input_string):
        return input_string.title()

    input_string = "hello world"
    print(title_case(input_string))
    # Output: Hello World

Problem 5: String Rotation

    def is_rotation(str1, str2):
        return len(str1) == len(str2) and str1 in str2 + str2

    str1 = "waterbottle"
    str2 = "erbottlewat"
    print(is_rotation(str1, str2))
    # Output: True

Problem 6: Replace Substring

    def replace_substring(input_string, target, replacement):
        return input_string.replace(target, replacement)

    input_string = "Hello, World!"
    target = "World"
    replacement = "Universe"
    print(replace_substring(input_string, target, replacement))
    # Output: Hello, Universe!

Problem 7: Acronym Generator

    def acronym_generator(input_string):
        words = input_string.split()
        acronym = "".join(word[0].upper() for word in words)
        return acronym

    input_string = "National Aeronautics and Space Administration"
    print(acronym_generator(input_string))
    # Output: NASA

Problem 8: Longest Substring

    def longest_substring(input_string):
        current_char = input_string[0]
        longest_substring = input_string[0]
        for char in input_string[1:]:
            if char == current_char[-1]:
                current_char += char
            else:
                current_char = char
            if len(current_char) > len(longest_substring):
                longest_substring = current_char
        return longest_substring

    input_string = "aabbaaaaabb"
    print(longest_substring(input_string))
    # Output: aaaaa
