Problem 8: Longest Substring

    def longest_substring(input_string):
        current_char = input_string[0]
        longest_substring = input_string[0]
        for char in input_string[1:]:
            if char == current_char[-1]:
                current_char += char
            else:
                current_char = char
            if len(current_char) > len(longest_substring):
                longest_substring = current_char
        return longest_substring

    input_string = "aabbaaaaabb"
    print(longest_substring(input_string))
    # Output: aaaaa

Longest Substring Without Repeating Characters:

Python Function:

def longest_substring(input_string):
    start = 0
    max_length = 0
    char_index = {}

    for index, char in enumerate(input_string):
        if char in char_index and char_index[char] >= start:
            start = char_index[char] + 1
        char_index[char] = index
        max_length = max(max_length, index - start + 1)

    return max_length

def generate_random_password(length):
    characters = string.ascii_letters + string.digits + string.punctuation
    password = ''.join(random.choice(characters) for _ in range(length))
    return password

Roman Numeral to Integer:

Python Function:

def roman_to_integer(roman):
    roman_numerals = {'I': 1, 'V': 5, 'X': 10, 'L': 50, 'C': 100, 'D': 500, 'M': 1000}
    num = 0
    prev_value = 0

    for numeral in reversed(roman):
        value = roman_numerals[numeral]
        if value < prev_value:
            num -= value
        else:
            num += value
        prev_value = value

    return num

Finding Largest Subarray with Sum Less Than or Equal to Target:

Python Function:

def find_largest_subarray(input_list, target):
    start, end = 0, 0
    current_sum = 0
    max_length = 0

    for end in range(len(input_list)):
        current_sum += input_list[end]

        while current_sum > target:
            current_sum -= input_list[start]
            start += 1

        max_length = max(max_length, end - start + 1)

    return max_length
