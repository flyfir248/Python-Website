Examples:

    
        list_of_sentences = ["This is a simple example example",
                             "This NLP exercise is fun and simple",
                             "Practice makes perfect"]

        # Get list of all words
        list_of_words = [word for sentence in list_of_sentences for word in sentence.split()]

        # Extract words not present in other sentences
        unique_words = [word for i, sentence in enumerate(list_of_sentences)
                        for word in sentence.split()
                        if word not in ' '.join(list_of_sentences[:i]+list_of_sentences[i+1:]).split()]

        # Count words in all the sentences using dictionary comprehension
        word_count = {}
        for sent in list_of_sentences:
            for word in set(sent.split()):
                if word not in word_count:
                    word_count[word] = sent.split().count(word)
                else:
                    word_count[word] += sent.split().count(word)

        # Displaying results
        console.log("List of all words:", list_of_words);
        console.log("Unique words:", unique_words);
        console.log("Word count:", word_count);

Python Exception Handling Examples

Code with and without Exception Handling:

    
        // Code without exception handling
        let num1 = parseInt(prompt("Please enter a number:"));
        let num2 = parseInt(prompt("Please enter another number:"));
        let func = prompt("Please enter a functionality:\t");

        if (func === "sum") {
            console.log(num1 + num2);
        } else if (func === "prod") {
            console.log(num1 * num2);
        } else if (func === "div") {
            console.log(num1 / num2);
        } else {
            console.log("Functionality not implemented");
        }

        // Code with exception handling
        try {
            num1 = parseInt(prompt("Please enter a number:"));
            num2 = parseInt(prompt("Please enter another number:"));
            func = prompt("Please enter a functionality:\t");

            if (func === "sum") {
                console.log(num1 + num2);
            } else if (func === "prod") {
                console.log(num1 * num2);
            } else if (func === "div") {
                if (num2 === 0) {
                    throw new Error("num2 cannot be 0 when functionality is div");
                }
                console.log(num1 / num2);
            } else {
                console.log("Functionality not implemented");
            }
        } catch (error) {
            console.error(error.message);
        }
