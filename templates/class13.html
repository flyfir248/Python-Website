16. Different Ways of Importing Functions from a Module

    # Importing the entire module
    import simplemath

    # Using the imported module's function
    result_circle = simplemath.areaOfCircle(2)
    print("Area of Circle Result:", result_circle)

    # Importing with aliasing
    import simplemath as sm

    # Using the alias for the function
    result_circle_alias = sm.areaOfCircle(3)
    print("Area of Circle (Aliased) Result:", result_circle_alias)

    # Importing specific functions
    from simplemath import areaOfCircle, areaOfRectangle

    # Using the directly imported functions
    result_rectangle = areaOfRectangle(2, 3)
    print("Area of Rectangle Result:", result_rectangle)

    # Importing all functions (not recommended)
    from simplemath import *

    # Using the imported functions
    result_square = areaOfSquare(3)
    print("Area of Square Result:", result_square)

17. List Comprehension

    # List comprehension to filter numbers divisible by 3
    li = [3, 4, 5, 67, 1, 2, 3, 9, 34]
    op_li = [i for i in li if i % 3 == 0]
    print("Filtered List (Divisible by 3):", op_li)

List Comprehension

Syntax:

        [expression for elements in iterator if filtering_condition]
            expression : data to append when the condition results in true
            element: item in each index of the iterator
            iterator : Any datatype that has multiple elements
            filtering_condition : Condition only when the expression is executed

Example:

    
    op_li = [i for i in li if i % 3 == 0]

    # expression --- i
    # element ---- i : Here expression and element are the same. This may not be the same always
    # iterator ---- li
    # filtering_condition ---- if i % 3 == 0

    print(op_li)
    # Output: [3, 3, 9]

List Comprehension

General Syntax:

        [expression for elements in iterator if filtering_condition]
            expression: data to append when the condition results in true
            element: item in each index of the iterator
            iterator: any datatype that has multiple elements
            filtering_condition: condition only when the expression is executed

Example 1:

    
    ip_li = [3, 4, 5, 67, 1, 2, 3, 9, 34]
    op_li = [[z, 'even' if z % 2 == 0 else 'odd'] for z in ip_li if z > 5]

    # expression --- [z, 'even' if z % 2 == 0 else 'odd']
    # element --- z
    # iterable ---- ip_li
    # filtering_condition ---- if z > 5

    print(op_li)
    # Output: [[67, 'odd'], [9, 'odd'], [34, 'even']]

Example 2:

    
    ip_li = [3, 4, 5]
    op_li = [y**3 for y in ip_li]

    # expression --- y**3
    # element --- y
    # iterable ---- ip_li
    # filtering_condition ---- No condition

    print(op_li)
    # Output: [27, 64, 125]

List Comprehension

General Syntax:

        [expression for elements in iterator if filtering_condition]
            expression: data to append when the condition results in true
            element: item in each index of the iterator
            iterator: any datatype that has multiple elements
            filtering_condition: condition only when the expression is executed

Example 1:

    
    # List comprehension to filter words not in stop_words and punctuations
    stop_words = ['the', 'of', 'at', 'or', 'in', 'and', 'by', 'to']
    punctuations = ['-', '.']
    filtered_words = [w for w in words if (w.lower() not in stop_words) and (w.lower() not in punctuations)]

    # expression ---- w
    # element ---- w
    # iterable ---- words
    # filtering_condition ---- if w.lower() not in stop_words + punctuations

    print(filtered_words)
    # Output: ['control', 'room', 'Bruhat', 'Bengaluru', 'Mahanagara', 'Palike', 'headquarters', 'nerve', 'centre', 'disaster', 'management', 'event', 'rain', 'flooding', 'appears', 'be', 'hit', 'absenteeism.']

Example 2:

    
    # Nested list comprehension for list flattening
    li = [[1, 2], [87, 5], [77, 7]]
    op = [j for i in li for j in i]

    # Expression ---- j
    # Elements ------ i and j
    # Iterator ------ li and i
    # Condition ----- No condition

    print(op)
    # Output: [1, 2, 87, 5, 77, 7]

    # List comprehension to extract words not present in other sentences
    list_of_sentences = ["This is a simple example ", "This NLP exercise is fun and simple", "Practice makes perfect"]
    op = [word for sentence in list_of_sentences for word in sentence.split()]

    print(op)
    # Output: ['This', 'is', 'a', 'simple', 'example', 'This', 'NLP', 'exercise', 'is', 'fun', 'and', 'simple', 'Practice', 'makes', 'perfect']

Example 3:

    
    # List comprehension to extract words not present in other sentences
    list_of_sentences = ["This is a simple example", "This NLP exercise is fun and simple", "Practice makes perfect"]
    res = []

    for i, sentence in enumerate(list_of_sentences):
        other_sent = list_of_sentences[:i] + list_of_sentences[i + 1:]
        word_in_other_sent = ' '.join(other_sent).split()

        for word in sentence.split():
            if word not in word_in_other_sent:
                res.append(word)

    print(res)
    # Output: ['a', 'example', 'NLP', 'exercise', 'fun', 'and', 'Practice', 'makes', 'perfect']
