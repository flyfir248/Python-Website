Difference between extend and append - Important for interviews:

    capital_cities_li = ['Bangalore', 'Hyderabad', 'Chennai', 'Mumbai', 'Delhi', 'Kolkata']

    print(capital_cities_li)
    # Output: ['Bangalore', 'Hyderabad', 'Chennai', 'Mumbai', 'Delhi', 'Kolkata']

    capital_cities_li.append('Ahmedabad')
    print(capital_cities_li)
    # Output: ['Bangalore', 'Hyderabad', 'Chennai', 'Mumbai', 'Delhi', 'Kolkata', 'Ahmedabad']

    new_list_of_capitals = ['Jaipur', 'Imphal', 'Bhopal']

    # Using append:
    # capital_cities_li.append(new_list_of_capitals)  # This will add the list as an element in the list.

    # Using extend:
    capital_cities_li.extend(new_list_of_capitals)  # This will add elements of the list to the original list.

    print(capital_cities_li)
    # Output: ['Bangalore', 'Hyderabad', 'Chennai', 'Mumbai', 'Delhi', 'Kolkata', 'Ahmedabad', 'Jaipur', 'Imphal', 'Bhopal']

    print(capital_cities_li[0])
    # Output: 'Bangalore'

    # Inserting into a multidimensional list:
    li = [1, 'Python', 22.6, 'Hello', 4]
    print(li)
    # Output: [1, 'Python', 22.6, 'Hello', 4]

    li.insert(2, 'Java')
    print(li)
    # Output: [1, 'Python', 'Java', 22.6, 'Hello', 4]

    li.insert(3, [6, 7, 8])
    print(li)
    # Output: [1, 'Python', 'Java', [6, 7, 8], 22.6, 'Hello', 4]

    # Pop and insert in a multidimensional list:
    li[3].insert(2, 'iiiiiiiiiiii')
    print(li)
    # Output: [1, 'Python', 'Java', [6, 7, 'iiiiiiiiiiii', 8], 22.6, 'Hello', 4]

    li.pop()
    print(li)
    # Output: [1, 'Python', 'Java', [6, 7, 'iiiiiiiiiiii', 8], 22.6, 'Hello']

    li.pop(2)
    print(li)
    # Output: [1, 'Python', [6, 7, 'iiiiiiiiiiii', 8], 22.6, 'Hello']

    # Output: <class 'list'>
    print(d)
    # Output: [2.45]

    print(type(c))
    # Output: <class 'str'>
    print(c)
    # Output: 2.45

    print(type(a))
    # Output: <class 'int'>
    print(a)
    # Output: 2

List and String Operations:

      # Not allowed because list does not have the function split
      # d.split()

      # Split is allowed on c because its datatype is string.
      c.split('.')
      # Output: ['2', '45']

      # Remove function removes the first occurrence of the value.
      li.remove(1)
      # Output: ['Python', 22.6, 'Hello', 4]

      # Count function returns the number of occurrences of a value in the list.
      print(li.count(6))
      # Output: 7
      print(li.count(4))
      # Output: 2

      # Reverse function reverses the list in place.
      li1.reverse()
      # Output: [6, 5, 4, 3, 2, 1]
      li2 = li2[::-1]
      # Output: [6, 5, 4, 3, 2, 1]

      # Sort function sorts the list in ascending order.
      li.sort()
      # Output: [1, 3, 4, 5, 22, 45, 76]

      # Sort function can also sort the list in descending order.
      li.sort(reverse=True)
      # Output: [76, 45, 22, 5, 4, 3, 1]

      # Sorted function does not modify the original list and returns a sorted list.
      sorted_st_marks = sorted(st_marks)
      # Output: [1, 4, 5, 12, 18, 19, 20, 21]

      # Sort in descending order using sorted function.
      st_marks.sort()
      # Output: [21, 20, 19, 18, 12, 5, 4, 1]

      # Sorting strings in a list.
      li = ['Python','Java','C++','c','c++','c--']
      li.sort()
      # Output: ['C++', 'Java', 'Python', 'c', 'c++', 'c--']

      # Extract even numbers from a list.
